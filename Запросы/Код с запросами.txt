-- Запрос 1: Вывести все продажи, которые успешно прошли проверку в системе "Честный Знак" (ГИС МТ)
-- Показывает умение фильтровать текстовые данные и связывать таблицы.
SELECT 
    П.[НомерДокумента], 
    П.[ДатаДокумента], 
    П.[ТипДокумента], 
    П.[Сумма], 
    К.[НазваниеОрганизации] AS [Покупатель],
    С.[НазваниеСклада] AS [Склад Отгрузки]
FROM [КнигаПродаж] П
JOIN [Контрагенты] К ON П.[КодКонтрагента] = К.[КодКонтрагента]
JOIN [Склады] С ON П.[КодСклада] = С.[КодСклада]
WHERE П.[СтатусГИСМТ] = N'Проверен в ГИС МТ';


-- Запрос 2: Вывести документы из Книги Покупок, сумма которых превышает 20 000 рублей
-- Демонстрирует фильтрацию по числовым диапазонам и сортировку.
SELECT 
    [НомерДокумента], 
    [ДатаДокумента], 
    [Сумма], 
    [ВходящийНомер]
FROM [КнигаПокупок]
WHERE [Сумма] > 20000.00
ORDER BY [Сумма] DESC;

-- Запрос 3: Посчитать общие суммы и количество продаж для каждого контрагента (клиента)
-- Показывает умение подводить финансовые итоги и группировать данные.

SELECT 
    К.[НазваниеОрганизации] AS [Контрагент],
    COUNT(П.[КодЗаписи]) AS [ВсегоДокументов],
    SUM(П.[Сумма]) AS [ОбщаяСуммаПродаж],
    AVG(П.[Сумма]) AS [СреднийЧекПродажи]
FROM [КнигаПродаж] П
JOIN [Контрагенты] К ON П.[КодКонтрагента] = К.[КодКонтрагента]
GROUP BY К.[НазваниеОрганизации];


-- Запрос 4: Посчитать общие затраты на закупки (Книга покупок) в разрезе видов операций
-- Помогает руководству пивзавода понять, на что уходит больше денег (на товары или услуги).

SELECT 
    О.[НазваниеОперации] AS [Тип Затрат],
    SUM(КП.[Сумма]) AS [ИтогоЗатрачено],
    COUNT(КП.[КодЗаписи]) AS [КоличествоПоставок]
FROM [КнигаПокупок] КП
JOIN [ВидыОпераций] О ON КП.[КодОперации] = О.[КодОперации]
GROUP BY О.[НазваниеОперации];

-- Запрос 5: Обновление данных (UPDATE)
-- Допустим, пришла информация из бухгалтерии, что по накладной '00БП-000003' статус счет-фактуры изменился.

UPDATE [КнигаПокупок]
SET [СтатусФактуры] = N'Проведен', 
    [Ответственный] = N'Старший бухгалтер'
WHERE [НомерДокумента] = '00БП-000003';


-- Запрос 6: Удаление данных (DELETE)
-- Демонстрирует умение удалять ошибочные или аннулированные записи. 
-- Удалим тестовую или ошибочную запись по конкретному номеру входящего документа.

DELETE FROM [КнигаПокупок]
WHERE [ВходящийНомер] = '165704' AND [Сумма] = 1209.60;
